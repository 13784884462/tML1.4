185  internal static void EnsureSyncedDependencyStability(TopoSort<LocalMod> synced, TopoSort<LocalMod> full)
186  {
187  var errored = new HashSet<LocalMod>();
188  var errorLog = new StringBuilder();
189 
190  foreach (var mod in synced.list)
191  {
192  var chains = new List<List<LocalMod>>();
193  //define recursive chain finding method
194  Action<LocalMod, Stack<LocalMod>> FindChains = null;
195  FindChains = (search, stack) =>
196  {
197  stack.Push(search);
198 
199  if (search.properties.side == ModSide.Both && stack.Count > 1)
200  {
201  if (stack.Count > 2)//direct Both -> Both references are ignored
202  chains.Add(stack.Reverse().ToList());
203  }
204  else
205  {//recursively build the chain, all entries in stack should be unsynced
206  foreach (var dep in full.Dependencies(search))
207  FindChains(dep, stack);
208  }
209 
210  stack.Pop();
211  };
212  FindChains(mod, new Stack<LocalMod>());
213 
214  if (chains.Count == 0)
215  continue;
216 
217  var syncedDependencies = synced.AllDependencies(mod);
218  foreach (var chain in chains)
219  if (!syncedDependencies.Contains(chain.Last()))
220  {
221  errored.Add(mod);
222  errorLog.AppendLine(mod + " indirectly depends on " + chain.Last() + " via " + String.Join(" -> ", chain));
223  }
224  }
225 
226  if (errored.Count > 0)
227  {
228  errorLog.AppendLine("Some of these mods may not exist on both client and server. Add a direct sort entries or weak references.");
229  throw new ModSortingException(errored, errorLog.ToString());
230  }

241  internal static List<LocalMod> Sort(ICollection<LocalMod> mods)
242  {
243  var preSorted = mods.OrderBy(mod => mod.Name).ToList();
244  var syncedSort = BuildSort(preSorted.Where(mod => mod.properties.side == ModSide.Both).ToList());
245  var fullSort = BuildSort(preSorted);

247
