261  {
262  ServicePointManager.Expect100Continue = false;

264  var values = new NameValueCollection
265  {
266  { "modloaderversion", ModLoader.versionedName },
267  { "platform", ModLoader.compressedPlatformRepresentation },
268  };
269  using (WebClient client = new WebClient())
270  {
271  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback((sender, certificate, chain, policyErrors) => { return true; });
272  client.UploadValuesCompleted += new UploadValuesCompletedEventHandler(UploadComplete);
273  client.UploadValuesAsync(new Uri(url), "POST", values);
274  }
275  }
276  catch (WebException e)
277  {
278  if (e.Status == WebExceptionStatus.Timeout)
279  {
280  SetHeading("Mod Browser OFFLINE (Busy)");
281  return;
282  }
283  if (e.Status == WebExceptionStatus.ProtocolError)
284  {
285  var resp = (HttpWebResponse)e.Response;
286  if (resp.StatusCode == HttpStatusCode.NotFound)
287  {
288  SetHeading("Mod Browser OFFLINE (404)");
289  return;
290  }
291  SetHeading("Mod Browser OFFLINE..");
292  return;
293  }
294  }

340  private void PopulateFromJSON(List<BuildProperties> modBuildProperties, string json)
341  {
342  try
343  {
344  JObject jsonObject = JObject.Parse(json);
345  JObject updateObject = (JObject)jsonObject["update"];
346  if (updateObject != null)
347  {
348  updateAvailable = true;
349  updateText = (string)updateObject["message"];
350  updateURL = (string)updateObject["url"];
351  }
352  JArray modlist = (JArray)jsonObject["modlist"];
353  foreach (JObject mod in modlist.Children<JObject>())
354  {
355  string displayname = (string)mod["displayname"];
356  string name = (string)mod["name"];
357  string version = (string)mod["version"];
358  string author = (string)mod["author"];
359  string download = (string)mod["download"];
360  int downloads = (int)mod["downloads"];
361  int hot = (int)mod["hot"]; // for now, hotness is just downloadsYesterday
362  string timeStamp = (string)mod["updateTimeStamp"];
363  bool exists = false;
364  bool update = false;
365  bool updateIsDowngrade = false;
366  foreach (BuildProperties bp in modBuildProperties)
367  {
368  if (bp.displayName.Equals(displayname))
369  {
370  exists = true;
371  if (!bp.version.Equals(new Version(version.Substring(1))))
372  {
373  update = true;
374  if (bp.version > new Version(version.Substring(1)))
375  {
376  updateIsDowngrade = true;
377  }
378  }
379  }
380  }
381  UIModDownloadItem modItem = new UIModDownloadItem(displayname, name, version, author, download, downloads, hot, timeStamp, update, updateIsDowngrade, exists);
382  modListAll._items.Add(modItem); //add directly to the underlying, SortList will repopulate it anyway
383  }
384  SortList();
385  }
386  catch (Exception e)
387  {
388  ErrorLogger.LogModBrowserException(e);
389  return;
390  }
