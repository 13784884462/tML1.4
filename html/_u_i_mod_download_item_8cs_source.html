335  public static string HumanTimeSpanString(DateTime yourDate)
336  {
337  var ts = new TimeSpan(DateTime.UtcNow.Ticks - yourDate.Ticks);
338  double delta = Math.Abs(ts.TotalSeconds);
339 
340  if (delta < 1 * MINUTE)
341  return ts.Seconds == 1 ? "1 second ago" : ts.Seconds + " seconds ago";
342 
343  if (delta < 2 * MINUTE)
344  return "1 minute ago";
345 
346  if (delta < 45 * MINUTE)
347  return ts.Minutes + " minutes ago";
348 
349  if (delta < 90 * MINUTE)
350  return "1 hour ago";
351 
352  if (delta < 24 * HOUR)
353  return ts.Hours + " hours ago";
354 
355  if (delta < 48 * HOUR)
356  return "1 day ago";
357 
358  if (delta < 30 * DAY)
359  return ts.Days + " days ago";
360 
361  if (delta < 12 * MONTH)
362  {
363  int months = Convert.ToInt32(Math.Floor((double)ts.Days / 30));
364  return months <= 1 ? "1 month ago" : months + " mnths ago";
365  }
366  else
367  {
368  int years = Convert.ToInt32(Math.Floor((double)ts.Days / 365));
369  return years <= 1 ? "1 year ago" : years + " years ago";
370  }
371  }
