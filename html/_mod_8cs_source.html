108  return;
109 
110  Interface.loadMods.SubProgressText = path;
111 
112  string extension = Path.GetExtension(path);
113  path = Path.ChangeExtension(path, null);
114  switch (extension)
115  {
116  case ".png":
117  case ".rawimg":
118  //png files need a seekable stream
119  LoadTexture(path, len, reader, extension == ".rawimg");
120  return;
121  case ".wav":
122  LoadWav(path, reader.ReadBytes(len));
123  return;
124  case ".mp3":
125  LoadMP3(path, reader.ReadBytes(len));
126  return;
127  case ".xnb":

129  {
130  LoadFont(path, reader.ReadBytes(len));
131  return;
132  }
133  if (path.StartsWith("Effects/"))
134  {
135  LoadEffect(path, reader);
136  return;
137  }
138  throw new ResourceLoadException(Language.GetTextValue("tModLoader.LoadErrorUnknownXNBFileHint", path));
139  }
140 
141  throw new ResourceLoadException($"Unknown streaming asset {path}{extension}. ");
