188

190  writer.Write(name);
191  writer.Write(version.ToString());
192 
193  // write file table
194  // file count
195  // file-entries:
196  // filename
197  // uncompressed file size
198  // compressed file size (stored size)
199  fileTable = files.Values.ToArray();
200  writer.Write(fileTable.Length);
201

205  throw new Exception($"compressedSize ({f.compressedSize}) != cachedBytes.Length ({f.cachedBytes.Length}): {f.name}");
206 
207  writer.Write(f.name);
208  writer.Write(f.size);
209  writer.Write(f.compressedSize);
210  }
211 
212  // write compressed files and update offsets
213  int offset = (int)fileStream.Position; // offset starts at end of file table
214  foreach (var f in fileTable) {
215  writer.Write(f.cachedBytes);
216
