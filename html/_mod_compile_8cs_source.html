500  return;
501 
502  bool generatePDB = mod.properties.includePDB && forWindows;
503  if (generatePDB && !windows)
504  {
505  Console.WriteLine(Language.GetTextValue("tModLoader.BuildErrorPDBWindowsOnly"));
506  generatePDB = false;
507  }
508 
509  //collect all dll references
510  var tempDir = Path.Combine(ModPath, "compile_temp");

512  var refs = new List<string>();
513 
514  //everything used to compile the tModLoader for the target platform
515  refs.AddRange(GetTerrariaReferences(tempDir, forWindows));
516

519 
520  //all dlls included in all referenced mods
521  foreach (var refMod in refMods)
522  {

528  {
529  path = Path.Combine(tempDir, refDll + ".dll");
530  File.WriteAllBytes(path, refMod.modFile.GetFile("lib/" + refDll + ".dll"));
531  refs.Add(path);
532  }
533  }

536  {
537  OutputAssembly = Path.Combine(tempDir, mod + ".dll"),
538  GenerateExecutable = false,
539  GenerateInMemory = false,

541  IncludeDebugInformation = generatePDB,
